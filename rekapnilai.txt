    <script>
        let rekapData = [];
        let nilaiColumns = []; // Menyimpan nama kolom nilai yang terdeteksi
        let bobotPerKolom = 0; // Bobot per kolom (dibagi rata)
        let customBobot = {}; // Menyimpan bobot custom jika user mengatur manual
        let isCustomBobot = false; // Flag apakah menggunakan bobot custom

        // Drag & Drop
        const uploadArea = document.getElementById('uploadArea');
        
        uploadArea.addEventListener('dragover', (e) => {
            e.preventDefault();
            uploadArea.classList.add('dragover');
        });

        uploadArea.addEventListener('dragleave', () => {
            uploadArea.classList.remove('dragover');
        });

        uploadArea.addEventListener('drop', (e) => {
            e.preventDefault();
            uploadArea.classList.remove('dragover');
            const file = e.dataTransfer.files[0];
            if (file) processExcel(file);
        });

        function handleFile(event) {
            const file = event.target.files[0];
            if (file) processExcel(file);
        }

        function processExcel(file) {
            const reader = new FileReader();
            reader.onload = function(e) {
                try {
                    const data = new Uint8Array(e.target.result);
                    const workbook = XLSX.read(data, { type: 'array' });
                    const firstSheet = workbook.Sheets[workbook.SheetNames[0]];
                    const jsonData = XLSX.utils.sheet_to_json(firstSheet);

                    if (jsonData.length === 0) {
                        alert('File Excel kosong!');
                        return;
                    }

                    processData(jsonData);
                    showTab('rekap');
                } catch (error) {
                    alert('Error membaca file: ' + error.message);
                }
            };
            reader.readAsArrayBuffer(file);
        }

        function processData(data) {
            if (data.length === 0) return;

            // Deteksi kolom nilai otomatis (selain No, Nama Siswa, Kelas)
            const excludeColumns = ['no', 'nama siswa', 'nama', 'kelas'];
            const firstRow = data[0];
            nilaiColumns = Object.keys(firstRow).filter(col => 
                !excludeColumns.includes(col.toLowerCase().trim())
            );

            if (nilaiColumns.length === 0) {
                alert('Tidak ada kolom nilai yang terdeteksi! Pastikan ada kolom selain No, Nama Siswa, dan Kelas.');
                return;
            }

            // Jika tidak ada custom bobot, set default rata
            if (!isCustomBobot) {
                bobotPerKolom = 100 / nilaiColumns.length;
                customBobot = {};
                nilaiColumns.forEach(col => {
                    customBobot[col] = bobotPerKolom;
                });
            }

            console.log('Kolom nilai terdeteksi:', nilaiColumns);
            console.log('Bobot:', customBobot);

            rekapData = data.map((row, index) => {
                // Ambil nilai dari setiap kolom nilai dan hitung total
                let totalNilai = 0;
                const nilaiPerKolom = {};
                
                nilaiColumns.forEach(col => {
                    const nilai = parseFloat(row[col] || 0);
                    nilaiPerKolom[col] = nilai;
                    const bobot = customBobot[col] || bobotPerKolom;
                    totalNilai += nilai * (bobot / 100);
                });

                const nilaiAkhir = Math.round(totalNilai * 10) / 10;

                return {
                    no: index + 1,
                    nama: row['Nama Siswa'] || row['Nama'] || row['nama'] || row['nama siswa'] || '-',
                    kelas: row['Kelas'] || row['kelas'] || '-',
                    ...nilaiPerKolom, // Spread semua nilai per kolom
                    nilaiAkhir: nilaiAkhir,
                    predikat: getPredikat(nilaiAkhir)
                };
            });

            // Sort by nilai akhir untuk ranking
            rekapData.sort((a, b) => b.nilaiAkhir - a.nilaiAkhir);
            rekapData.forEach((item, index) => {
                item.ranking = index + 1;
            });

            renderTable();
            updateStats();
            updateBobotDisplay();
        }

        function getPredikat(nilai) {
            if (nilai >= 85) return 'A';
            if (nilai >= 70) return 'B';
            if (nilai >= 55) return 'C';
            return 'D';
        }

        function renderTable() {
            const tbody = document.getElementById('tableBody');
            const thead = document.querySelector('#rekapTable thead tr');
            
            if (rekapData.length === 0) {
                // Reset ke default header
                thead.innerHTML = `
                    <th>No</th>
                    <th>Nama Siswa</th>
                    <th>Kelas</th>
                    <th>Tugas</th>
                    <th>UH</th>
                    <th>UTS</th>
                    <th>UAS</th>
                    <th>Nilai Akhir</th>
                    <th>Predikat</th>
                    <th>Ranking</th>
                `;
                tbody.innerHTML = `
                    <tr>
                        <td colspan="10" class="empty-state">
                            <div class="empty-state-icon">📊</div>
                            <div>Belum ada data. Upload file Excel untuk memulai.</div>
                        </td>
                    </tr>
                `;
                return;
            }

            // Generate header dinamis berdasarkan kolom nilai yang terdeteksi
            let headerHTML = `
                <th>No</th>
                <th>Nama Siswa</th>
                <th>Kelas</th>
            `;
            
            nilaiColumns.forEach(col => {
                headerHTML += `<th>${col}</th>`;
            });
            
            headerHTML += `
                <th>Nilai Akhir</th>
                <th>Predikat</th>
                <th>Ranking</th>
            `;
            
            thead.innerHTML = headerHTML;

            // Generate body dengan nilai dinamis
            tbody.innerHTML = rekapData.map((row, index) => {
                const nilaiClass = row.nilaiAkhir >= 85 ? 'nilai-tinggi' : row.nilaiAkhir < 60 ? 'nilai-rendah' : '';
                
                let nilaiCells = '';
                nilaiColumns.forEach(col => {
                    nilaiCells += `<td>${row[col] || 0}</td>`;
                });

                return `
                    <tr>
                        <td>${index + 1}</td>
                        <td><strong>${row.nama}</strong></td>
                        <td>${row.kelas}</td>
                        ${nilaiCells}
                        <td><span class="nilai-akhir ${nilaiClass}">${row.nilaiAkhir}</span></td>
                        <td><span class="predikat predikat-${row.predikat}">${row.predikat}</span></td>
                        <td><span class="ranking">${row.ranking}</span></td>
                    </tr>
                `;
            }).join('');

            document.getElementById('totalRows').textContent = rekapData.length;
            document.getElementById('totalData').textContent = rekapData.length;
        }

        function updateStats() {
            if (rekapData.length === 0) return;

            const nilaiArray = rekapData.map(r => r.nilaiAkhir);
            const max = Math.max(...nilaiArray);
            const min = Math.min(...nilaiArray);
            const avg = (nilaiArray.reduce((a, b) => a + b, 0) / nilaiArray.length).toFixed(1);

            document.getElementById('nilaiMax').textContent = max;
            document.getElementById('nilaiMin').textContent = min;
            document.getElementById('rataKelas').textContent = avg;
        }

        function filterTable() {
            const searchValue = document.getElementById('searchInput').value.toLowerCase();
            const rows = document.querySelectorAll('#tableBody tr');
            let visibleCount = 0;

            rows.forEach(row => {
                const text = row.textContent.toLowerCase();
                if (text.includes(searchValue)) {
                    row.style.display = '';
                    visibleCount++;
                } else {
                    row.style.display = 'none';
                }
            });

            document.getElementById('totalRows').textContent = visibleCount;
        }

        function showTab(tab) {
            document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
            event.target.closest('.tab')?.classList.add('active');

            if (tab === 'upload') {
                document.getElementById('uploadSection').style.display = 'block';
                document.getElementById('rekapSection').style.display = 'none';
            } else if (tab === 'rekap') {
                document.getElementById('uploadSection').style.display = 'none';
                document.getElementById('rekapSection').style.display = 'block';
            }
        }

        function downloadTemplate() {
            // Template 1: Contoh dengan 4 kolom nilai
            const template1 = [

        function exportToExcel() {
            if (rekapData.length === 0) {
                alert('Tidak ada data untuk di-export!');
                return;
            }

            const exportData = rekapData.map(row => {
                const baseData = {
                    'No': row.no,
                    'Nama Siswa': row.nama,
                    'Kelas': row.kelas
                };

                // Tambahkan semua kolom nilai
                nilaiColumns.forEach(col => {
                    baseData[col] = row[col];
                });

                // Tambahkan nilai akhir, predikat, ranking
                baseData['Nilai Akhir'] = row.nilaiAkhir;
                baseData['Predikat'] = row.predikat;
                baseData['Ranking'] = row.ranking;

                return baseData;
            });

            const ws = XLSX.utils.json_to_sheet(exportData);
            const wb = XLSX.utils.book_new();
            XLSX.utils.book_append_sheet(wb, ws, 'Rekap Nilai');
            XLSX.writeFile(wb, `rekap_nilai_${new Date().toISOString().split('T')[0]}.xlsx`);
        }

        function resetData() {
            if (confirm('Yakin ingin menghapus semua data?')) {
                rekapData = [];
                nilaiColumns = [];
                customBobot = {};
                isCustomBobot = false;
                renderTable();
                showTab('upload');
            }
        }

        function updateBobotDisplay() {
            const bobotInfo = document.getElementById('bobotInfo');
            const bobotDisplay = document.getElementById('bobotDisplay');
            
            if (nilaiColumns.length === 0) {
                bobotInfo.style.display = 'none';
                return;
            }

            bobotInfo.style.display = 'block';
            
            let displayText = '';
            nilaiColumns.forEach((col, index) => {
                const bobot = customBobot[col] || bobotPerKolom;
                displayText += `<strong>${col}:</strong> ${bobot.toFixed(1)}%`;
                if (index < nilaiColumns.length - 1) {
                    displayText += ' • ';
                }
            });
            
            if (isCustomBobot) {
                displayText += ' <span style="color: #3b82f6; font-weight: 600;">(Custom)</span>';
            } else {
                displayText += ' <span style="color: #718096;">(Otomatis - dibagi rata)</span>';
            }
            
            bobotDisplay.innerHTML = displayText;
        }

        function openBobotModal() {
            const modal = document.getElementById('bobotModal');
            const inputsContainer = document.getElementById('bobotInputs');
            
            // Generate input untuk setiap kolom nilai
            inputsContainer.innerHTML = nilaiColumns.map(col => {
                const currentBobot = customBobot[col] || bobotPerKolom;
                return `
                    <div style="display: flex; align-items: center; gap: 15px; padding: 12px; background: #f7fafc; border-radius: 8px;">
                        <label style="flex: 1; font-weight: 500; color: #2d3748;">${col}</label>
                        <input 
                            type="number" 
                            id="bobot-${col}" 
                            value="${currentBobot.toFixed(1)}" 
                            min="0" 
                            max="100" 
                            step="0.1"
                            onkeyup="updateTotalBobot()"
                            oninput="updateTotalBobot()"
                            style="width: 100px; padding: 8px; border: 2px solid #cbd5e0; border-radius: 6px; text-align: center; font-weight: 600;"
                        >
                        <span style="color: #718096; font-weight: 500;">%</span>
                    </div>
                `;
            }).join('');
            
            modal.style.display = 'flex';
            updateTotalBobot();
        }

        function closeBobotModal() {
            document.getElementById('bobotModal').style.display = 'none';
        }

        function updateTotalBobot() {
            let total = 0;
            nilaiColumns.forEach(col => {
                const input = document.getElementById(`bobot-${col}`);
                if (input) {
                    total += parseFloat(input.value) || 0;
                }
            });
            
            const display = document.getElementById('totalBobotDisplay');
            display.textContent = total.toFixed(1) + '%';
            
            if (Math.abs(total - 100) < 0.1) {
                display.style.color = '#059669';
            } else {
                display.style.color = '#dc2626';
            }
        }

        function resetBobot() {
            const bobotRata = 100 / nilaiColumns.length;
            nilaiColumns.forEach(col => {
                const input = document.getElementById(`bobot-${col}`);
                if (input) {
                    input.value = bobotRata.toFixed(1);
                }
            });
            updateTotalBobot();
        }

        function saveBobotSettings() {
            let total = 0;
            const newBobot = {};
            
            nilaiColumns.forEach(col => {
                const input = document.getElementById(`bobot-${col}`);
                if (input) {
                    const nilai = parseFloat(input.value) || 0;
                    newBobot[col] = nilai;
                    total += nilai;
                }
            });
            
            if (Math.abs(total - 100) > 0.1) {
                alert(`Total bobot harus 100%!\nSaat ini: ${total.toFixed(1)}%`);
                return;
            }
            
            customBobot = newBobot;
            isCustomBobot = true;
            
            // Hitung ulang semua nilai
            if (rekapData.length > 0) {
                rekapData = rekapData.map(row => {
                    let totalNilai = 0;
                    nilaiColumns.forEach(col => {
                        const nilai = row[col] || 0;
                        totalNilai += nilai * (customBobot[col] / 100);
                    });
                    
                    const nilaiAkhir = Math.round(totalNilai * 10) / 10;
                    
                    return {
                        ...row,
                        nilaiAkhir: nilaiAkhir,
                        predikat: getPredikat(nilaiAkhir)
                    };
                });
                
                // Re-sort dan update ranking
                rekapData.sort((a, b) => b.nilaiAkhir - a.nilaiAkhir);
                rekapData.forEach((item, index) => {
                    item.ranking = index + 1;
                });
                
                renderTable();
                updateStats();
            }
            
            updateBobotDisplay();
            closeBobotModal();
            
            alert('✅ Bobot berhasil disimpan dan nilai akhir telah dihitung ulang!');
        }
